```python
from difflib import SequenceMatcher

def title_similarity(title1, title2):
    if pd.isna(title1) or pd.isna(title2):
        return 0.0
    return SequenceMatcher(None, str(title1).lower(), str(title2).lower()).ratio()

df['title_match'] = df.apply(lambda x: title_similarity(x['SupplierTitle'], x['AmazonTitle']), axis=1)
```

### **STAGE 3: STRICT EAN Matching (CRITICAL)**

(KEEP THIS SCRIPT SNIPPET AS-IS)

```python
def is_valid_ean(ean):
    """Check if EAN is a valid barcode (not empty, nan, None, etc.)"""
    if pd.isna(ean):
        return False
    ean_str = str(ean).strip()
    return ean_str not in ['nan', '', 'None', 'NaN', '0', '-']

def is_exact_ean_match(row):
    """Returns True ONLY if BOTH EANs are valid AND they match exactly"""
    ean_sup = str(row['EAN']).strip()
    ean_amz = str(row['EAN_OnPage']).strip()
    
    # Both must be valid
    if not is_valid_ean(ean_sup) or not is_valid_ean(ean_amz):
        return False
    
    # Must match exactly
    return ean_sup == ean_amz

df['is_exact_ean'] = df.apply(is_exact_ean_match, axis=1)
```

### **STAGE 4: Pack Size Extraction & Profit Recalculation**

(KEEP THIS SCRIPT SNIPPET AS-IS)

```python
import re

def extract_quantity(title):
    """Extract pack size from product title. Defaults to 1."""
    if pd.isna(title):
        return 1.0
    title = str(title).lower()
    
    patterns = [
        r'pack of (\d+)',
        r'set of (\d+)',
        r'\b(\d+)\s*pack\b',
        r'\b(\d+)\s*pk\b',
        r'(\d+)\s*pcs\b',
        r'(\d+)\s*pieces?\b',
        r'(\d+)\s*pairs?\b',
        r'\bx\s*(\d+)\b',
        r'\((\d+)\s*pack\)',
        r'\(pack of (\d+)\)',
        r'\b(\d+)\s*rolls?\b',
        r'\b(\d+)\s*piece\b',
    ]
    
    for pat in patterns:
        match = re.search(pat, title)
        if match:
            qty = float(match.group(1))
            if qty > 1 and qty < 500:  # Sanity check
                return qty
    return 1.0

df['Sup_Qty'] = df['SupplierTitle'].apply(extract_quantity)
df['Amz_Qty'] = df['AmazonTitle'].apply(extract_quantity)
df['Qty_Ratio'] = df['Amz_Qty'] / df['Sup_Qty']

def recalculate_profit(row):
    """
    Adjust profit based on quantity ratio.
    If Amazon sells a 6-pack and Supplier sells singles,
    we need to buy 6 units, so: Adjusted_Profit = Original - (Cost * (Ratio - 1))
    """
    try:
        original_profit = float(row['NetProfit'])
        supplier_cost = float(row['SupplierPrice_incVAT'])
        ratio = row['Qty_Ratio']
        adjustment = supplier_cost * (ratio - 1)
        return original_profit - adjustment
    except:
        return 0.0

df['Adjusted_Profit'] = df.apply(recalculate_profit, axis=1)
```

### **STAGE 5: Product Categorization**

(KEEP THIS SCRIPT SNIPPET AS-IS)

```python
def categorize(row):
    if row['is_exact_ean']:  # STRICT: Both EANs valid AND matching
        return 'EXACT_EAN_MATCH'
    elif row['title_match'] >= 0.50:
        return 'HIGH_LIKELIHOOD'
    elif row['title_match'] >= 0.30:
        return 'MODERATE_CONFIDENCE'
    else:
        return 'UNCERTAIN'

df['category'] = df.apply(categorize, axis=1)

def pack_verdict(row):
    if row['Qty_Ratio'] == 1.0:
        return "1:1 Match"
    elif row['Qty_Ratio'] > 1.0:
        if row['Adjusted_Profit'] > 0:
            return f"BUNDLE ({int(row['Qty_Ratio'])}x) - OK"
        else:
            return f"BUNDLE ({int(row['Qty_Ratio'])}x) - LOSS"
    else:
        if row['Adjusted_Profit'] > 0:
            return f"SPLIT (1/{int(1/row['Qty_Ratio'])}) - OK"
        else:
            return "SPLIT - LOSS"

df['Pack_Verdict'] = df.apply(pack_verdict, axis=1)
```

## 🆕 STAGE 6 (ADDED): MANUAL PACK-SIZE VERIFICATION FOR NON-MATCHING EAN SHORTLISTS

IMPORTANT: This stage is REQUIRED for any row that you plan to output under **HIGH LIKELIHOOD** and/or **NEEDS VERIFICATION** where EANs do NOT match or are missing.

### ✅ RULE: THIS MUST BE DONE MANUALLY (NO SCRIPT)

* Do NOT use regex, scripted extraction, or automated pack parsing for this step.
* You must thoroughly analyze the titles/info as a human would and document the evidence.

### Manual protocol (do this row-by-row for shortlisted non-EAN rows)

For each candidate:

1. Read SupplierTitle and AmazonTitle carefully and identify any count/pack cues:

   * Examples of cues: “5 bags”, “30 bags”, “6 packs of 40”, “240 bags total”, “3 x”, “45 bags”, “(20)”, “2 x”, “pk10”, “10 pieces”, “set of”.
2. Decide which case applies (must choose one):

   * **1:1 Match** (same count/pack)
   * **Supplier smaller pack** (Amazon pack is larger; requires buying multiple supplier units)
   * **Supplier larger pack** (split scenario; supplier multipack while Amazon is single)
   * **Ambiguous** (insufficient clarity; DO NOT shortlist; route to HIGH LIKELIHOOD – FILTERED OUT if it was considered)
3. If Supplier smaller pack:

   * Compute **Required Supplier Units (RSU)** = (Amazon total units) / (Supplier units)
   * If RSU is not a clean integer, treat as **Ambiguous** unless there is an explicit “total” that resolves it.
4. Compute profitability after manual pack correction:

   * **Manual_Adjusted_Profit (approx) = NetProfit − SupplierPrice_incVAT × (RSU − 1)**
5. Decision:

   * If Manual_Adjusted_Profit ≤ 0 → EXCLUDE from recommendation tables and include in **HIGH LIKELIHOOD — FILTERED OUT (Audit)** with the pack math noted.
   * If Manual_Adjusted_Profit > 0 → eligible (subject to Sales>0 and NetProfit>0 filters).
6. Write evidence into the report row:

   * “Pack Verdict” must reflect the manual conclusion (e.g., “BUNDLE (6x) - OK”)
   * “Adjusted Profit (approx)” must reflect Manual_Adjusted_Profit (approx) for non-EAN shortlisted rows.
   * “Key Match Evidence” must cite the exact title cues (numbers/phrases).
   * “Key Risks / Notes” must mention EAN mismatch/missing and any variant/assorted ambiguity.

### Mandatory scenarios you must catch (examples)

* Supplier: “TIDYZ WHEELY BIN LINERS 5 BAGS 300L”
  Amazon: “Tidyz 30 Extra Large Wheelie Bin Liners … 300L”
  → RSU = 30/5 = 6; cost must be multiplied by 6 (via profit adjustment).
* Supplier: “TIDYZ PEDAL BIN LINERS 40 WHITE … 15L”
  Amazon: “Tidyz 6 Packs Of 40 … 240 Bags Total”
  → RSU = 240/40 = 6.
* Supplier: “BACOFOIL ZIPPER BAGS … 15 PACK …”
  Amazon: “Bacofoil 3 x … (45 Bags)”
  → RSU = 45/15 = 3.

## 🆕 STAGE 6B (ADDED): MANUAL VERIFICATION FOR EXACT-EAN MATCHES (TO PREVENT “SILENT DROPS”)

Sometimes scripted pack parsing can incorrectly penalize true exact-EAN rows (especially where titles contain dimensions like “9 x 9 inch” or multiple numbers). Therefore:

### When to trigger Stage 6B (Exact-EAN Review Trigger)

For any row where `is_exact_ean == True`, perform a quick **manual title sanity check** IF ANY of these are present:

* multiple numbers in either title
* an “x” pattern between two numbers (often dimensions)
* words like: pack, pk, set, bundle, bags, liners, rolls, pcs/pieces, pairs, total
* “assorted”, “variety”, “mixed”, “refill”, “replacement”, “compatible”

### Manual protocol (exact-EAN sanity)

For each triggered exact-EAN row:

1. Confirm product type and variant alignment from titles (size/capacity/dimensions/colour/model).
2. Confirm whether numbers are **pack counts** or **dimensions/capacity** (document your interpretation).
3. If you find a clear pack mismatch or clear variant mismatch:

   * EXCLUDE from recommendation tables
   * INCLUDE in **VERIFIED (Exact EAN) — FILTERED OUT (Audit)** with explicit reason.
4. If it looks consistent (or only ambiguous but not contradictory):

   * Keep in VERIFIED recommendation table, and add a note if ambiguity remains (e.g., “numbers appear to be dimensions; pack assumed 1:1”).

```python
from difflib import SequenceMatcher

def title_similarity(title1, title2):
    if pd.isna(title1) or pd.isna(title2):
        return 0.0
    return SequenceMatcher(None, str(title1).lower(), str(title2).lower()).ratio()
```

### **STAGE 3: STRICT EAN Matching (CRITICAL)**

(KEEP THIS SCRIPT SNIPPET AS-IS)

```python
def is_valid_ean(ean):
    """Check if EAN is a valid barcode (not empty, nan, None, etc.)"""
    if pd.isna(ean):
        return False
    ean_str = str(ean).strip()
    return ean_str not in ['nan', '', 'None', 'NaN', '0', '-']

def is_exact_ean_match(row):
    """Returns True ONLY if BOTH EANs are valid AND they match exactly"""
    ean_sup = str(row['EAN']).strip()
    ean_amz = str(row['EAN_OnPage']).strip()
    
    # Both must be valid
    if not is_valid_ean(ean_sup) or not is_valid_ean(ean_amz):
        return False
    
    # Must match exactly
    return ean_sup == ean_amz

df['is_exact_ean'] = df.apply(is_exact_ean_match, axis=1)
```

### **STAGE 4: Pack Size Extraction & Profit Recalculation**

(KEEP THIS SCRIPT SNIPPET AS-IS)

```python
import re

def extract_quantity(title):
    """Extract pack size from product title. Defaults to 1."""
    if pd.isna(title):
        return 1.0
    title = str(title).lower()
    
    patterns = [
        r'pack of (\d+)',
        r'set of (\d+)',
        r'\b(\d+)\s*pack\b',
        r'\b(\d+)\s*pk\b',
        r'(\d+)\s*pcs\b',
        r'(\d+)\s*pieces?\b',
        r'(\d+)\s*pairs?\b',
        r'\bx\s*(\d+)\b',
        r'\((\d+)\s*pack\)',
        r'\(pack of (\d+)\)',
        r'\b(\d+)\s*rolls?\b',
        r'\b(\d+)\s*piece\b',
    ]
    
    for pat in patterns:
        match = re.search(pat, title)
        if match:
            qty = float(match.group(1))
            if qty > 1 and qty < 500:  # Sanity check
                return qty
    return 1.0

df['Sup_Qty'] = df['SupplierTitle'].apply(extract_quantity)
df['Amz_Qty'] = df['AmazonTitle'].apply(extract_quantity)
df['Qty_Ratio'] = df['Amz_Qty'] / df['Sup_Qty']

def recalculate_profit(row):
    """
    Adjust profit based on quantity ratio.
    If Amazon sells a 6-pack and Supplier sells singles,
    we need to buy 6 units, so: Adjusted_Profit = Original - (Cost * (Ratio - 1))
    """
    try:
        original_profit = float(row['NetProfit'])
        supplier_cost = float(row['SupplierPrice_incVAT'])
        ratio = row['Qty_Ratio']
        adjustment = supplier_cost * (ratio - 1)
        return original_profit - adjustment
    except:
        return 0.0

df['Adjusted_Profit'] = df.apply(recalculate_profit, axis=1)
```

```python
def categorize(row):

def pack_verdict(row):

    elif row['Qty_Ratio'] > 1.0:
        if row['Adjusted_Profit'] > 0:

        else:

    else:
        if row['Adjusted_Profit'] > 0:
            return f"SPLIT (1/{int(1/row['Qty_Ratio'])}) - OK"
        else:
            return "SPLIT - LOSS"

df['Pack_Verdict'] = df.apply(pack_verdict, axis=1)
```

3. **Decide the pack relationship (must choose one)**

   * **1:1 Match** (same count/pack)
   * **Supplier smaller pack** (Amazon pack is larger; requires buying multiple supplier units)
   * **Supplier larger pack** (split scenario; supplier multipack while Amazon is single)
   * **Ambiguous** (insufficient clarity → EXCLUDED; route to audit)

4. **If Supplier smaller pack (bundle case), compute RSU**

   * Compute **Required Supplier Units (RSU)** = (Amazon total units) / (Supplier units)
   * RSU should be a clean integer when counts are explicit.
   * If RSU is not a clean integer:

     * Only accept it if the title explicitly provides a “total” that resolves the ambiguity.
     * Otherwise mark Ambiguous → EXCLUDED to audit.

5. **Compute profitability after manual pack correction**

   * **Manual_Adjusted_Profit (approx) = NetProfit − SupplierPrice_incVAT × (RSU − 1)**
   * This formula mirrors the “additional units cost” logic and is intended to catch the most common bundle trap.

6. **Decision rules (non-EAN rows)**

   * If **Manual_Adjusted_Profit (approx) ≤ 0**:

     * EXCLUDE from recommendation tables
     * INCLUDE in **HIGH LIKELIHOOD — FILTERED OUT (Audit)** (or the relevant audit section)
     * Clearly write `EXCLUDED:` and the RSU math in the row.
   * If **Manual_Adjusted_Profit (approx) > 0**:

     * The row may be eligible for recommendation tables, but only if it also meets:

       * Sales > 0 and NetProfit > 0, and
       * No obvious variant/category mismatch found in Step 1.

7. **Write evidence into the report row**

   * “Pack Verdict” must reflect the manual conclusion (e.g., “BUNDLE (6x) - OK”, “BUNDLE (6x) - LOSS”, “1:1 Match”, “SPLIT - Ambiguous”).
   * “Adjusted Profit (approx)” must reflect Manual_Adjusted_Profit (approx) for non-EAN shortlisted rows.
   * “Key Match Evidence” must cite the exact title cues (numbers/phrases) you used.
   * “Key Risks / Notes” must mention:

     * EAN mismatch/missing, and
     * any remaining uncertainty (variant ambiguity, units ambiguous, likely dimension vs count confusion).

### Mandatory scenarios you must catch (examples)

* Supplier: “TIDYZ WHEELY BIN LINERS 5 BAGS 300L”
  Amazon: “Tidyz 30 Extra Large Wheelie Bin Liners … 300L”
  → RSU = 30/5 = 6; cost must be multiplied by 6 (via profit adjustment).
* Supplier: “TIDYZ PEDAL BIN LINERS 40 WHITE … 15L”
  Amazon: “Tidyz 6 Packs Of 40 … 240 Bags Total”
  → RSU = 240/40 = 6.
* Supplier: “BACOFOIL ZIPPER BAGS … 15 PACK …”
  Amazon: “Bacofoil 3 x … (45 Bags)”
  → RSU = 45/15 = 3.

```python
def is_valid_ean(ean):
    """Check if EAN is a valid barcode (not empty, nan, None, etc.)"""
    if pd.isna(ean):
        return False
    ean_str = str(ean).strip()
    return ean_str not in ['nan', '', 'None', 'NaN', '0', '-']

def is_exact_ean_match(row):
    """Returns True ONLY if BOTH EANs are valid AND they match exactly"""
    ean_sup = str(row['EAN']).strip()
    ean_amz = str(row['EAN_OnPage']).strip()
    
    # Both must be valid
    if not is_valid_ean(ean_sup) or not is_valid_ean(ean_amz):
        return False
    
    # Must match exactly
    return ean_sup == ean_amz

df['is_exact_ean'] = df.apply(is_exact_ean_match, axis=1)
```

```python
        r'(\d+)\s*pcs\b',
        r'(\d+)\s*pieces?\b',
        r'(\d+)\s*pairs?\b',
        r'\bx\s*(\d+)\b',
        r'\((\d+)\s*pack\)',
        r'\(pack of (\d+)\)',
        r'\b(\d+)\s*rolls?\b',
        r'\b(\d+)\s*piece\b',
    ]
    
    for pat in patterns:
        match = re.search(pat, title)
        if match:
            qty = float(match.group(1))
            if qty > 1 and qty < 500:  # Sanity check
                return qty
    return 1.0

df['Sup_Qty'] = df['SupplierTitle'].apply(extract_quantity)
df['Amz_Qty'] = df['AmazonTitle'].apply(extract_quantity)
df['Qty_Ratio'] = df['Amz_Qty'] / df['Sup_Qty']

def recalculate_profit(row):
    """
    Adjust profit based on quantity ratio.
    If Amazon sells a 6-pack and Supplier sells singles,
    we need to buy 6 units, so: Adjusted_Profit = Original - (Cost * (Ratio - 1))
    """
    try:
        original_profit = float(row['NetProfit'])
        supplier_cost = float(row['SupplierPrice_incVAT'])
        ratio = row['Qty_Ratio']
        adjustment = supplier_cost * (ratio - 1)
        return original_profit - adjustment
    except:
        return 0.0

df['Adjusted_Profit'] = df.apply(recalculate_profit, axis=1)
```

## 🆕 STAGE 6 (ADDED): MANUAL PACK-SIZE VERIFICATION FOR NON-MATCHING EAN SHORTLISTS

IMPORTANT: This stage is REQUIRED for any row that you plan to output under **HIGH LIKELIHOOD** and/or **NEEDS VERIFICATION** where EANs do NOT match or are missing.

### ✅ RULE: THIS MUST BE DONE MANUALLY (NO SCRIPT)

* Do NOT use regex, scripted extraction, or automated pack parsing for this step.
* You must thoroughly analyze the titles/info as a human would and document the evidence.

### Manual protocol (do this row-by-row for shortlisted non-EAN rows)

For each candidate:

1. Read SupplierTitle and AmazonTitle carefully and identify any count/pack cues:

   * Examples of cues: “5 bags”, “30 bags”, “6 packs of 40”, “240 bags total”, “3 x”, “45 bags”, “(20)”, “2 x”, “pk10”, “10 pieces”, “set of”.
2. Decide which case applies (must choose one):

   * **1:1 Match** (same count/pack)
   * **Supplier smaller pack** (Amazon pack is larger; requires buying multiple supplier units)
   * **Supplier larger pack** (split scenario; supplier multipack while Amazon is single)
   * **Ambiguous** (insufficient clarity; DO NOT shortlist; route to HIGH LIKELIHOOD – FILTERED OUT if it was considered)
3. If Supplier smaller pack:

   * Compute **Required Supplier Units (RSU)** = (Amazon total units) / (Supplier units)
   * If RSU is not a clean integer, treat as **Ambiguous** unless there is an explicit “total” that resolves it.
4. Compute profitability after manual pack correction:

   * **Manual_Adjusted_Profit (approx) = NetProfit − SupplierPrice_incVAT × (RSU − 1)**
5. Decision:

   * If Manual_Adjusted_Profit ≤ 0 → EXCLUDE from recommendation tables and include in **HIGH LIKELIHOOD — FILTERED OUT (Audit)** with the pack math noted.
   * If Manual_Adjusted_Profit > 0 → eligible (subject to Sales>0 and NetProfit>0 filters).
6. Write evidence into the report row:

   * “Pack Verdict” must reflect the manual conclusion (e.g., “BUNDLE (6x) - OK”)
   * “Adjusted Profit (approx)” must reflect Manual_Adjusted_Profit (approx) for non-EAN shortlisted rows.
   * “Key Match Evidence” must cite the exact title cues (numbers/phrases).
   * “Key Risks / Notes” must mention EAN mismatch/missing and any variant/assorted ambiguity.

  → RSU = 30/5 = 6; cost must be multiplied by 6 (via profit adjustment).

import re

def clean_to_digits(x):
    if pd.isna(x):
        return ''
    s = str(x).strip()
    # If scientific notation appears, treat as corrupted (cannot reliably recover original)
    if 'e+' in s.lower() or 'e' in s.lower():

    return re.sub(r'\D', '', s)

def is_valid_ean(ean):
    """Check if EAN is a valid barcode (not empty, nan, None, etc.)"""
    if pd.isna(ean):
        return False
    ean_str = str(ean).strip()
    return ean_str not in ['nan', '', 'None', 'NaN', '0', '-']

def is_exact_ean_match(row):
    """Returns True ONLY if BOTH EANs are valid AND they match exactly"""
    ean_sup = str(row['EAN']).strip()
    ean_amz = str(row['EAN_OnPage']).strip()
    
    # Both must be valid
    if not is_valid_ean(ean_sup) or not is_valid_ean(ean_amz):
        return False
    
    # Must match exactly
    return ean_sup == ean_amz

df['is_exact_ean'] = df.apply(is_exact_ean_match, axis=1)

🆕 STAGE 3B (ADDED): STRICT BARCODE VALIDITY + CHECKSUM (SOURCE-OF-TRUTH FOR “EXACT EAN”)

IMPORTANT: Use is_exact_ean_strict (not is_exact_ean) for:
	•	VERIFIED pool selection
	•	any “Exact EAN Match” language

def gtin_checksum_ok(digits: str) -> bool:
    if not digits.isdigit():
        return False
    n = len(digits)
    if n not in (8, 12, 13, 14):
        return False
    
    # For checksum: compute check digit for all but last digit
    body = digits[:-1]
    check = int(digits[-1])

    # Weighting rules depend on length; for GTIN-12/13/14 and EAN-8 use alternating 3/1 from right
    body_rev = list(map(int, body[::-1]))
    total = 0
    for i, d in enumerate(body_rev, start=1):
        total += d * (3 if i % 2 == 1 else 1)
    calc = (10 - (total % 10)) % 10
    return calc == check

def is_strict_valid_barcode(digits: str) -> bool:
    if not isinstance(digits, str):
        return False
    if not digits.isdigit():
        return False
    if len(digits) not in (8, 12, 13, 14):
        return False
    # Reject obvious placeholder/corrupted patterns (heavy trailing zeros)
    if re.search(r'0{6,}$', digits):  # 6+ zeros at end is usually suspicious in your datasets
        return False
    return gtin_checksum_ok(digits)

df['EAN_strict_valid'] = df['EAN_digits'].apply(is_strict_valid_barcode)
df['EAN_OnPage_strict_valid'] = df['EAN_OnPage_digits'].apply(is_strict_valid_barcode)

df['is_exact_ean_strict'] = (
    df['EAN_strict_valid']

STAGE 4: Pack Size Extraction & Profit Recalculation

(KEEP THIS SCRIPT SNIPPET AS-IS)

import re

def extract_quantity(title):
    """Extract pack size from product title. Defaults to 1."""
    if pd.isna(title):
        return 1.0
    title = str(title).lower()
    
    patterns = [
        r'pack of (\d+)',
        r'set of (\d+)',
        r'\b(\d+)\s*pack\b',
        r'\b(\d+)\s*pk\b',
        r'(\d+)\s*pcs\b',
        r'(\d+)\s*pieces?\b',
        r'(\d+)\s*pairs?\b',
        r'\bx\s*(\d+)\b',
        r'$begin:math:text$\(\\d\+\)\\s\*pack$end:math:text$',
        r'$begin:math:text$pack of \(\\d\+\)$end:math:text$',
        r'\b(\d+)\s*rolls?\b',
        r'\b(\d+)\s*piece\b',
    ]
    
    for pat in patterns:
        match = re.search(pat, title)
        if match:
            qty = float(match.group(1))
            if qty > 1 and qty < 500:  # Sanity check
                return qty
    return 1.0

df['Sup_Qty'] = df['SupplierTitle'].apply(extract_quantity)
df['Amz_Qty'] = df['AmazonTitle'].apply(extract_quantity)
df['Qty_Ratio'] = df['Amz_Qty'] / df['Sup_Qty']

def recalculate_profit(row):
    """
    Adjust profit based on quantity ratio.
    If Amazon sells a 6-pack and Supplier sells singles,
    we need to buy 6 units, so: Adjusted_Profit = Original - (Cost * (Ratio - 1))
    """
    try:
        original_profit = float(row['NetProfit'])
        supplier_cost = float(row['SupplierPrice_incVAT'])
        ratio = row['Qty_Ratio']
        adjustment = supplier_cost * (ratio - 1)
        return original_profit - adjustment
    except:
        return 0.0

df['Adjusted_Profit'] = df.apply(recalculate_profit, axis=1)

    elif row['Qty_Ratio'] > 1.0:

        else:
            return "SPLIT - LOSS"

df['Pack_Verdict'] = df.apply(pack_verdict, axis=1)
